# reverse-nodes-in-k-group
``` cpp
Example:
Given this linked list: 1->2->3->4->5
For k = 2, you should return: 2->1->4->3->5
For k = 3, you should return: 3->2->1->4->5
```

**思路：递归解决，注意到每一个group的head都需要连接到下一个group的end。也就说可以利用递归函数返回下一个group的end。这个end就是该group经过reverse之后的head**

``` cpp
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        
        ListNode* last = head;
        
        for (int i = 0; i < k; i++) {
            if (last == nullptr) return head;
            last = last->next;
        }
        
        ListNode* newHead = reverse(head, last);
        head->next = reverseKGroup(last, k);
        return newHead;
    }
    
    ListNode* reverse(ListNode* head, ListNode* last) {
        ListNode* prev = nullptr;
        ListNode* cur = head;
        ListNode* next = nullptr;
        while (cur != last) {
            next = cur->next;
            cur->next = prev;
            prev = cur;
            cur = next;
        }
        return prev;
    }
};
```

# swap nodes in pairs

``` cpp
Given a linked list, swap every two adjacent nodes and return its head.
Example:
Given 1->2->3->4, you should return the list as 2->1->4->3.
```

- cur表示当前节点，next表示下一个节点，cur->next = next->next, next->next = cur
- 然后迭代，问题是如何改变2->1->3->4中的1->next，记录1为prev节点

``` cpp
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        ListNode* dummy = new ListNode(0);
        dummy->next = head;
        ListNode* prev = dummy;
        ListNode* cur = nullptr;
        ListNode* next = nullptr;
        
        while ((cur = prev->next) != nullptr && (next = cur->next) != nullptr) {
	        cur->next = next->next;
	        next->next = cur;
	        prev->next = next;
	        prev = cur;
        }
        return dummy->next;
    }
};

class Solution1 {
public:
    ListNode* swapPairs(ListNode* head) {
        
        ListNode** prev = &head;
        ListNode* cur = nullptr;
        ListNode* next = nullptr;
        
        while((cur = *prev) != nullptr && (next = cur->next) != nullptr) {
            cur->next = next->next;
            next->next = cur;
            *prev = next;
            prev = &cur->next;
        }
        
        return head;
    }    
};
```

# sort list

** 使用归并排序对list进行排序，时间复杂度O(nlogn) **

``` cpp
Sort a linked list in O(n log n) time using constant space complexity.
Example 1:
Input: 4->2->1->3
Output: 1->2->3->4
Example 2:

Input: -1->5->3->4->0
Output: -1->0->3->4->5
```

``` cpp
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if (head == nullptr || head->next == nullptr) return head;
        ListNode* mid = findMid(head);
        ListNode* next = mid->next;
        mid->next = nullptr;
        
        ListNode* head1 = sortList(head);
        ListNode* head2 = sortList(next);
        
        return merge(head1, head2);
    }
    
    ListNode* findMid(ListNode* head) {
        auto slow = head;
        auto fast = head;
        
        while(fast != nullptr && fast->next != nullptr && fast->next->next != nullptr) {
            slow = slow->next;
            fast = fast->next->next;
        }
        
        return slow;
    }
    
    ListNode* merge(ListNode* head1, ListNode* head2) {
        ListNode dummy(0);
        auto tail = &dummy;
        while (head1 != nullptr && head2 != nullptr) {
            if (head1->val < head2->val) {
                tail->next = head1;
                head1 = head1->next;
            } else {
                tail->next = head2;
                head2 = head2->next;
            }
            tail = tail->next;
        }        
        if (head1 != nullptr) {
            tail->next = head1;
        }
        if (head2 != nullptr) {
            tail->next = head2;
        }        
        return dummy.next;
    }
};
```

# rotate list

``` cpp
Example 1:
Input: 1->2->3->4->5->NULL, k = 2
Output: 4->5->1->2->3->NULL
Explanation:
rotate 1 steps to the right: 5->1->2->3->4->NULL
rotate 2 steps to the right: 4->5->1->2->3->NULL

Example 2:
Input: 0->1->2->NULL, k = 4
Output: 2->0->1->NULL
Explanation:
rotate 1 steps to the right: 2->0->1->NULL
rotate 2 steps to the right: 1->2->0->NULL
rotate 3 steps to the right: 0->1->2->NULL
rotate 4 steps to the right: 2->0->1->NULL
```

- 计算链表长度，并将其连成环
- 接着从head走len - k%len步，并断开链表

``` cpp
class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if (head == nullptr) return head;
        ListNode* node1 = head;
        ListNode* node2 = head;
        int len = 1;
        while(node1->next != nullptr) {
            node1 = node1->next;
            len++;
        }
        node1->next = head;
        k = len - k%len;
        while(--k) {
            node2 = node2->next;
        }              
        node1 = node2->next;
        node2->next = nullptr;
        return node1;
    }
};
```

# split linked list in parts
**思路：这道题思路还是简单的，直接撸就完事了**
``` cpp
class Solution {
public:
    vector<ListNode*> splitListToParts(ListNode* root, int k) { 
        if (root == nullptr) return vector<ListNode*>(k);
        vector<ListNode*> results;
        int len = 0;
        ListNode* node = root;
        while (node != nullptr) {
            ++len;
            node = node->next;
        }
        int numOfSeg = len/k;
        int extraOfSeg = len%k;
        
        node = root;
        while(k--) {
            int num = numOfSeg + (extraOfSeg-- > 0 ? 1 : 0);
            ListNode* newHead=node;
            while (--num && node != nullptr) {
                node = node->next;
            }
            if (node != nullptr) {
                ListNode* next = node->next;
                node->next = nullptr;
                node = next;
            }
            results.push_back(newHead);
        }
        return results;
    }
};
```

# remove duplicate from sorted list ii

**删除所有重复的节点**
``` cpp
Example 1:
Input: 1->2->3->3->4->4->5
Output: 1->2->5

Example 2:
Input: 1->1->1->2->3
Output: 2->3
```
- 重点在于记录重复节点前面的一个节点
- 要么使用dummy节点，要么使用二级指针记录，否则head节点需要多一次判断
``` cpp
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        ListNode* node = head;
        ListNode** prev = &head;
        while(node && node->next) {
            if (node->val == node->next->val) {
                auto val = node->val;
                while(node && val == node->val) {
                    auto del = node;
                    node = node->next;
                    delete del;
                }
                *prev = node;
            } else {
                prev = &node->next;
                node = node->next;  
            }          
        }
        return head;
    }
};
```

# revers linked list ii

``` cpp
Reverse a linked list from position m to n. Do it in one-pass.
Note: 1 ≤ m ≤ n ≤ length of list.
Example:
Input: 1->2->3->4->5->NULL, m = 2, n = 4
Output: 1->4->3->2->5->NULL
```

- 逆序部分链表
- 链表的题目一定会遇到head需要改变时该怎么办？一种最直接的方法是：采用if语句判断
- 优秀的方法：采用dummy节点获知二级指针

``` cpp
class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int m, int n) {
        ListNode** prev = &head;
        ListNode* cur = head;
        ListNode* next = cur;
        
        int k = n - m;
        while(--m) {
            prev = &cur->next;
            cur = cur->next;
        }
        while(k--) {
            next = cur->next;
            cur->next = next->next;
            next->next = *prev;
            *prev = next;
        }
        return head;      
    }
};
```

# partition list

**链表的部分排序**

``` cpp
Example:
Input: head = 1->4->3->2->5->2, x = 3
Output: 1->2->2->4->3->5
```
- 相当于快速排序的一次partition操作，去partition不同的是，这里要求所以小于x的值都在>=x的值之前
- 需要两个dummy节点
``` cpp
class Solution {
public:
    ListNode* partition(ListNode* head, int x) {
        ListNode less(0), great(0);
        ListNode* pLess = &less;
        ListNode* pGreat = &great;
        while(head) {
            if(head->val < x) {
                pLess = pLess->next = head;
            } else {
                pGreat = pGreat->next =head;
            }
            head = head->next;
        }
        pGreat->next = nullptr;
        pLess->next = great.next;   
        return less.next;
    }
};
```

# merge k sorted list

``` cpp
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
    	// 小顶堆
        auto cmp = [](auto const* lval, auto const* rval){ return lval->val > rval->val;};
        std::priority_queue<ListNode*, std::vector<ListNode*>, decltype(cmp)> pq(cmp); 
        for (auto list : lists) {
            if (list != nullptr) pq.push(list);
        }
        ListNode* pHead = nullptr;
        ListNode* pNode = pHead;
        //记录head节点
        if (!pq.empty()) {
            pHead = pq.top();  
            pNode = pHead;
            pq.pop();
            if(pNode->next) pq.push(pNode->next);         
        }
        while (!pq.empty()) {
            pNode->next = pq.top();
            pNode = pNode->next;
            pq.pop();
            if (pNode->next) pq.push(pNode->next);
        }
        return pHead;
    }
};
```

# linked list cycle ii

**思路：快慢指针找到相交点，然后根据**
- l为环形之前的长度，s为换的入口到交点的长度，r为环长度
- slow指针走过的距离 = l+s
- fast指针走过的距离 = l+nr+s
- 2*(l+s)=l+nr+s  ------> l = nr - s
在交点处，将slow指针放回原点，然后两个指针开始跑，当slow==fast时，说明已经到达交点
``` cpp
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        if (head == nullptr || head->next == nullptr) return nullptr;
        auto slow = head, fast = head;
        
        while (fast != nullptr && fast->next != nullptr ) {
            fast = fast->next->next;
            slow = slow->next;
            if (slow == fast) break;
        }
        
        if (fast == nullptr) return nullptr; 
        slow = head;
        
        while (fast != slow && fast && slow) {
            slow = slow->next;
            fast = fast->next;
        }
        return fast;
    }
};
```

``` cpp
class Solution {
public:
	RandomListNode *copyRandomList(RandomListNode *head) {
		if (head == nullptr) {
			return nullptr;
		}
		//利用node的next字段保存当前node的copy
		//利用copy的next字段保存当前node的next

			}

		RandomListNode* newNode = newHead;

			newNode->next = node->next;

	}
```
